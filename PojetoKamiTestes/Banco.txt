CREATE DATABASE kamikami;
USE kamikami;

CREATE TABLE usuario (
    id INT AUTO_INCREMENT PRIMARY KEY,
    cpf VARCHAR(14) UNIQUE NOT NULL,
    nome VARCHAR(100) NOT NULL,
    senha VARCHAR(255) NOT NULL,
    criado_em DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE pedido (
    id INT AUTO_INCREMENT PRIMARY KEY,
    forma_pagamento VARCHAR(50) NOT NULL,
    data_pedido DATETIME DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(20) NOT NULL DEFAULT 'Em preparação'
);

CREATE TABLE itens_pedido (
    id INT AUTO_INCREMENT PRIMARY KEY,
    pedido_id INT NOT NULL,
    nome VARCHAR(100) NOT NULL,
    quantidade INT NOT NULL,
    valor INT NOT NULL,
    FOREIGN KEY (pedido_id) REFERENCES pedido(id) 
);

SELECT p.id AS '# Pedido', 
p.data_pedido AS 'Data', 
SUM(ip.valor) AS 'Valor Total' 
FROM itens_pedido ip INNER JOIN pedido p ON ip.pedido_id = p.id 
GROUP BY ip.pedido_id 
ORDER BY p.data_pedido DESC;
